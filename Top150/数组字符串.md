#### 合并两个有序数组
``` cpp
class Solution {
public:
    void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {
        if(n==0) return;
        int n1=m-1,n2=n-1;
        n=nums1.size()-1;
        while(n1>=0&&n2>=0)
        {
            if(nums1[n1]>=nums2[n2]) nums1[n--]=nums1[n1--];
            else nums1[n--]=nums2[n2--];
        }
        while(n1>=0) nums1[n--]=nums1[n1--];
        while(n2>=0) nums1[n--]=nums2[n2--];
    }
};
```

#### 删除有序数组中的重复项
``` cpp
class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int dest=0,cur=0;
        while(cur<nums.size())
        {
            nums[dest]=nums[cur];
            while(cur<nums.size()&&nums[dest]==nums[cur])
                cur++;
            dest++;
        }
        return dest;
    }
};
```

#### 多数元素
``` cpp
class Solution {
public:
    int majorityElement(vector<int>& nums) {
        int key=nums[0];
        int count=1;
        for(int i=1;i<nums.size();i++)
        {
            if(nums[i]!=key)
            {
                count--;
                if(count==0)
                {
                    key=nums[i];
                    count=1;
                }
            }
            else count++;
        }
        return key;
    }
};
```

#### 买卖股票的最佳时机
``` cpp
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int n=prices.size();
        vector<int> arr(n);
        arr[n-1]=prices[n-1];
        for(int i=n-2;i>=0;i--)
        {
            arr[i]=max(arr[i+1],prices[i]);
        }
        int ret=0;
        for(int i=0;i<n;i++)
        {
            ret=max(ret,arr[i]-prices[i]);
        }
        return max(ret,0);
    }
};
```

``` cpp
class Solution {
public:
    bool canJump(vector<int>& nums) {
        int ret=0,n=nums.size();
        for(int i=0;i<n;i++)
        {
            if(ret>=i)
            ret=max(ret,i+nums[i]);
        }
        return ret>=n-1;
    }
};
```

``` cpp
class Solution {

        int n=nums.size();
        int cur=0,prev=0,ret=0;
        for(int i=0;i<n-1;i++)
        {
            if(i+nums[i]>cur) cur=i+nums[i];
            if(i==prev)
            {
                prev=cur;
                ret++;
            }
```

``` cpp
    RandomizedSet() {

    }
    
    bool insert(int val) {
        if(hash.count(val)) return false;
        int n=v.size();
        v.push_back(val);
        hash[val]=n;
        return true;
    }
    
    bool remove(int val) {
        if(!hash.count(val)) return false;
        int n=hash[val];
        v[n]=v.back();
        hash[v.back()]=n;
        v.pop_back();
```

``` cpp
class Solution {
public:
    int dfs(vector<int>& g, vector<int>& c,int begin)
    {
        int n=g.size();
        int gas=g[begin]-c[begin];
            if(gas<0) return begin+1;
        int j;
        for( j=begin+1;j%n!=begin;j++)
        {
            int jj=j%n;
            gas+=g[jj];
            gas-=c[jj];
            if(gas<0) return j+1;
        }
        return begin;
    }
    int canCompleteCircuit(vector<int>& gas, vector<int>& cost) {
        int n=gas.size();
        int i=0;
        while(i<n)
        {
            int j=dfs(gas,cost,i);
            if(i!=j) i=j;
            else return i;
        }
        return -1;
    }
};
```

``` cpp
class Solution {
public:
    int candy(vector<int>& ratings) {
        int n=ratings.size();
        vector<int> left(n);
        vector<int> right(n);
        int ret=0;
        for(int i=0;i<n;i++)
        {
            if(i>0&&ratings[i]>ratings[i-1]) left[i]=left[i-1]+1;
            else left[i]=1;
        }
        for(int i=n-1;i>=0;i--)
        {
            if(i<n-1&&ratings[i]>ratings[i+1]) right[i]=right[i+1]+1;
            else right[i]=1;
            ret+= max(left[i],right[i]);
        }
        return ret;
    }
};
```

``` cpp
class Solution {
public:
    int trap(vector<int>& height) {
        int n=height.size();
        vector<int> arr1(n);
        vector<int> arr2(n);
        for(int i=1;i<n;i++)
        {
            arr1[i]=max(arr1[i-1],height[i-1]);
        }
        for(int i=n-2;i>=0;i--)
        {
            arr2[i]=max(arr2[i+1],height[i+1]);
        }
        int ret=0;
        for(int i=0;i<n;i++)
        {
            ret+=max(min(arr1[i],arr2[i])-height[i],0);
        }
        return ret;
    }
};
```

#### 罗马数字转整数
``` cpp
class Solution {
public:
    int romanToInt(string s) {
        int n=s.size();
        unordered_map<char,int> hash;
        hash['I']=1,hash['V']=5,hash['X']=10,hash['L']=50,hash['C']=100,hash['D']=500,hash['M']=1000;
        int ret=0;
        for(int i=0;i<n;i++)
        {
            if(i<n-1&&hash[s[i]]<hash[s[i+1]])
            {
                ret-=hash[s[i]];
            }
            else ret+=hash[s[i]];
        }
        return ret;

    }
};
```
