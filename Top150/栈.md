### 有效的括号
``` cpp
class Solution {
public:
#define XXX(st,c1,c2)   if(ch==c1) st.push(ch); else if(ch==c2){if(st.empty()||st.top()!=c1) return false; st.pop(); }
    bool isValid(string s) {
        stack<char> st;
        for(auto& ch:s)
        {
            XXX(st,'(',')')
            XXX(st,'[',']')
            XXX(st,'{','}')
        }
        return st.empty();
    }
};
```
