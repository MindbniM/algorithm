### 环形链表
``` cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    bool hasCycle(ListNode *head) {
        if(head==nullptr) return false;
        ListNode* p=head,*cur=head;
        do 
        {
            p=p->next;
            cur=cur->next;
            if(cur) cur=cur->next;
        }while(cur&&p!=cur);
        if(cur==nullptr) return false;
        return true;
    }
};
```
