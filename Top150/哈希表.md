### 赎金信
``` cpp
class Solution {
public:
    bool canConstruct(string ransomNote, string magazine) {
        unordered_map<char,int> hash;
        for(auto& i:magazine) hash[i]++;
        for(auto& c:ransomNote)
        {
            if(hash.count(c))
            {
                hash[c]--;
                if(hash[c]==0) hash.erase(c);
            }
            else return false;
        }
        return true;
    }
};
```
### 同构字符串
``` cpp
class Solution {
public:
    bool isIsomorphic(string s, string t) {
        if(s.size()!=t.size()) return false;
        unordered_map<char,char> hash1;
        unordered_map<char,char> hash2;
        for(int i=0;i<s.size();i++)
        {
            if(hash1.count(s[i])||hash2.count(t[i]))
            {
                if(t[i]!=hash1[s[i]]||s[i]!=hash2[t[i]]) return false;
            }
            else
            {
                hash1[s[i]]=t[i];
                hash2[t[i]]=s[i];
            }
        }
        return true;
    }
};
```
### 单词规律
``` cpp
class Solution {
public:
    bool wordPattern(string pattern, string s) {
        unordered_map<char,string> hash;
        unordered_map<string,char> hash1;
        size_t j=0;
        for(int i=0;i<pattern.size();i++)
        {
            int n=s.find(' ',j+1);
            string temp;
            if(n!=string::npos) temp=s.substr(j,n-j);
            else temp=s.substr(j);
            //cout<<temp<<endl;
            j=n+1;
            if(hash.count(pattern[i])&&hash[pattern[i]]!=temp) return false;
            else if(hash1.count(temp)&&hash1[temp]!=pattern[i]) return false;
            else 
            {
                hash[pattern[i]]=temp;
                hash1[temp]=pattern[i];
            }
        }
        if(j!=0) return false;
        return true;
    }
};
```

``` cpp
        sort(s.begin(),s.end());
        sort(t.begin(),t.end());
        return s==t;
```

### 字母异位词分组
``` cpp
class Solution {
public:
    vector<vector<string>> groupAnagrams(vector<string>& strs) {
        int n=strs.size();
        unordered_map<string,vector<string>> h;
        for(auto& str:strs)
        {
            string s(str);
            sort(s.begin(),s.end());
            h[s].push_back(str);
        }
        vector<vector<string>> ret;
        for(auto[_,v]:h)
        {
            ret.push_back(move(v));
        }
        return ret;
    }
};
```
### 两数之和
``` cpp
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int,int> hash;
        for(int i=0;i<nums.size();i++)
        {
            if(hash.count(target-nums[i]))
            {
                return {i,hash[target-nums[i]]};
            }
            hash[nums[i]]=i;
        }
        return {}; 
    }
};
```
### 快乐数
``` cpp
class Solution {
public:
    int getHappy(int n)
    {
        int num=0;
        while(n)
        {
            int i=n%10;
            num+=i*i;
            n/=10;
        }
        return num;
    }
    bool isHappy(int n) {
        set<int> hash;
        hash.insert(n);
        while(n!=1)
        {
            n=getHappy(n);
            if(hash.count(n)) return false;
            hash.insert(n);
        }
        return true;
    }
};
```
