### 验证回文串
``` cpp
class Solution {
public:
    bool isPalindrome(string s) {
        string temp;
        for(int i=0;i<s.size();i++)
        {
            if(s[i]>='A'&&s[i]<='Z') temp+=s[i]+32;
            else if(s[i]>='a'&&s[i]<='z') temp+=s[i];
            else if(s[i]>='0'&&s[i]<='9') temp+=s[i];
        }
        string t=temp;
        reverse(t.begin(),t.end());
        return t==temp;
    }
};
```
### 判断子序列
``` cpp
class Solution {
public:
    bool isSubsequence(string s, string t) {
        if(s=="") return true;
        int ret=0;
        for(auto& c:t)
        {
            if(c==s[ret])
            {
                ret++;
                if(ret==s.size()) return true;
            }
        }
        return false;
    }
};
```
### 两数之和 II - 输入有序数组
``` cpp
class Solution {
public:
    int binay(int i,int j,vector<int>& n,int num)
    {
        while(i<j)
        {
            int mid=i+(j-i)/2;
            if(n[mid]<num) i=mid+1;
            else if(n[mid]>num) j=mid;
            else return mid;
        }
        return i;
    }
    vector<int> twoSum(vector<int>& numbers, int target) {
        vector<int> v(2);
        int n=numbers.size();
        for(int i=0;i<n;i++)
        {
            int temp=target-numbers[i];
            int index=binay(i+1,n-1,numbers,temp);
            if(numbers[index]==temp)
            {
                v[0]=i+1,v[1]=index+1;
                return v;
            }
        }
        return v;
    }
};
```
