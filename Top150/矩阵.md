### 有效的数独
``` cpp
class Solution {
public:
    int arr1[10][10]={0};
    int arr2[10][10]={0};
    int arr3[10][10]={0};
    bool isValidSudoku(vector<vector<char>>& board) {
        for (int i = 0; i < board.size(); i++)
        {
            for (int j = 0; j < board[i].size(); j++)
            {
                int val = board[i][j] - '0';
                if (val >0)
                {
                    arr1[i][val]++;
                    if(arr1[i][val]>=2) return false;
                    arr2[j][val]++;
                    if(arr2[j][val]>=2) return false;
                    int n=0;
                    if(i>=3&&i<6) n=3;
                    else if(i>=6) n=6;
                    arr3[j/3+n][val]++;
                    if(arr3[j/3+n][val]>=2) return false;
                }
            }
        }
        return true;
    }
};
```
