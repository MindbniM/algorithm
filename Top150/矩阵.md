### 有效的数独
``` cpp
class Solution {
public:
    int arr1[10][10]={0};
    int arr2[10][10]={0};
    int arr3[10][10]={0};
    bool isValidSudoku(vector<vector<char>>& board) {
        for (int i = 0; i < board.size(); i++)
        {
            for (int j = 0; j < board[i].size(); j++)
            {
                int val = board[i][j] - '0';
                if (val >0)
                {
                    arr1[i][val]++;
                    if(arr1[i][val]>=2) return false;
                    arr2[j][val]++;
                    if(arr2[j][val]>=2) return false;
                    int n=0;
                    if(i>=3&&i<6) n=3;
                    else if(i>=6) n=6;
                    arr3[j/3+n][val]++;
                    if(arr3[j/3+n][val]>=2) return false;
                }
            }
        }
        return true;
    }
};
```
### 螺旋矩阵
``` cpp
class Solution {
public:
    vector<int> spiralOrder(vector<vector<int>>& matrix) {
        int n=matrix.size(),m=matrix[0].size();
        int w=0,s=n-1,a=0,d=m-1;
        vector<int> ret(m*n);
        int index=0;
        while(1)
        {
            for(int i=a;i<=d;i++) ret[index++]=matrix[w][i];
            if(++w>s) break;
            for(int i=w;i<=s;i++) ret[index++]=matrix[i][d];
            if(--d<a) break;
            for(int i=d;i>=a;i--) ret[index++]=matrix[s][i];
            if(--s<w) break;
            for(int i=s;i>=w;i--) ret[index++]=matrix[i][a];
            if(++a>d) break;
        }
        return ret;
    }
};
```
### 旋转图像
``` cpp
class Solution {
public:
    void rotate(vector<vector<int>>& matrix) {
        int n=matrix.size()-1;
        for(int i=0;i<=n/2;i++)
        {
            for(int j=i;j<n-i;j++)
            {
                int n1=matrix[i][j];
                int n2=matrix[j][n-i];
                int n3=matrix[n-i][n-j];
                int n4=matrix[n-j][i];
                matrix[i][j]=n4;
                matrix[j][n-i]=n1;
                matrix[n-i][n-j]=n2;
                matrix[n-j][i]=n3;
            }
        }
    }
};
```
